	\begin{algorithm}
		\caption{投影寻踪法}
		\begin{algorithmic}[1]
			\Require 数据集$D$，投影次数$T$，学习率$\eta$，正则化参数$\lambda$
			\Ensure 投影后的数据集$D'$
			\State 初始化投影矩阵$W$为$d \times d$的单位矩阵
			\For{$t = 1$ \textbf{to} $T$}
			\State $W \gets W - \eta \cdot (I - W)$
			\State $D' \gets D \cdot W$
			\State \textbf{if} {达到停止条件} \textbf{then} \textbf{break} \textbf{end if}
			\EndFor
			\State \Return $D'$
		\end{algorithmic}
	\end{algorithm}
