\begin{document}
	
	\begin{algorithm}
		\caption{K-Means聚类算法}
		\begin{algorithmic}[1]
			\Require 数据集$D$，簇的数量$K$，最大迭代次数$MaxIter$
			\Ensure 每个数据点的簇分配和簇中心
			\State 随机选择$K$个数据点作为初始簇中心
			\For{$iter = 1$ \textbf{to} $MaxIter$}
			\For{每个数据点$x$ \textbf{in} $D$}
			\State 将$x$分配到最近的簇中心
			\EndFor
			\For{每个簇$C$ \textbf{in} $K$}
			\State 更新簇$C$的中心为簇内所有点的均值
			\EndFor
			\If{簇中心没有变化}
			\State \textbf{break}
			\EndIf
			\EndFor
			\State \Return 簇分配和簇中心
		\end{algorithmic}
	\end{algorithm}
