	\begin{algorithm}
		\caption{岭回归}
		\begin{algorithmic}[1]
			\Require 数据集$D$，正则化参数$\lambda$
			\Ensure 岭回归模型参数$\beta$
			\State 初始化参数$\beta$为0
			\State 设置学习率$\alpha$和正则化参数$\lambda$
			\While{未达到停止条件}
			\State 计算预测值$\hat{y} = X\beta$
			\State 计算误差项$e = y - \hat{y}$
			\State 计算梯度$\nabla_{\beta} = -2X^T e + 2\lambda \beta$
			\State 更新参数$\beta \leftarrow \beta - \alpha \nabla_{\beta}$
			\EndWhile
			\State \Return $\beta$
		\end{algorithmic}
	\end{algorithm}
